  /**
  - Calculates the full SAR Balance Sheet.
  - 
  - @param eligibleAvgBalance  L863  — reference eligible average balance
  - @param sarBalance          C885  — eligible SAR balance
  - @param statReserveRate     0.07  for 7%
  - @param vofAnnualRate       0.0375 for 3.75%
  - @param dpfAnnualRate       0.0005 for 0.05%
  - @param niiShareRate        0.88  for 88%
  - @param vatRate             0.15  for 15%
  - @param daysInPeriod        e.g. 28
  - @param daysInYear          e.g. 365
  - @return BalanceSheetResult holding every intermediate and final computed value
    */
    static BalanceSheetResult calculate(
    double eligibleAvgBalance,
    double sarBalance,
    double statReserveRate,
    double vofAnnualRate,
    double dpfAnnualRate,
    double niiShareRate,
    double vatRate,
    int    daysInPeriod,
    int    daysInYear) {
    
    // ––––– Before Exceptions (C column) –––––
    double c_statReserve    = sarBalance * statReserveRate;
    double c_eligibleBalNet = sarBalance - c_statReserve;
    double c_vof            = (c_eligibleBalNet * vofAnnualRate / daysInYear) * daysInPeriod;
    double c_dpf            = (sarBalance        * dpfAnnualRate / daysInYear) * daysInPeriod;
    double c_hssShare       = c_vof - c_dpf;
    double c_totalNII       = c_hssShare * niiShareRate;
    double c_vatCost        = c_hssShare * vatRate;
    double c_claimFromSABB  = c_totalNII + c_hssShare + c_vatCost;
    
    // ––––– Alinma Exemption / NIL NII (D column) –––––
    double d_statReserve    = sarBalance * statReserveRate;
    double d_eligibleBalNet = sarBalance - d_statReserve;
    double d_vof            = (d_eligibleBalNet * vofAnnualRate / daysInYear) * daysInPeriod;
    double d_dpf            = (sarBalance        * dpfAnnualRate / daysInYear) * daysInPeriod;
    double d_hssShare       = d_vof - d_dpf;
    double d_totalNII       = d_hssShare * niiShareRate;
    double d_vatCost        = d_hssShare * vatRate;
    double d_claimFromSABB  = d_totalNII + d_hssShare + d_vatCost;
    
    // ––––– After Exceptions (F column) = C + D –––––
    double f_sarBalance     = sarBalance       + sarBalance;
    double f_eligibleBalNet = c_eligibleBalNet  + d_eligibleBalNet;
    double f_vof            = c_vof             + d_vof;
    double f_dpf            = c_dpf             + d_dpf;
    double f_hssShare       = c_hssShare        + d_hssShare;
    double f_totalNII       = c_totalNII        + d_totalNII;
    double f_vatCost        = c_vatCost         + d_vatCost;
    double f_claimFromSABB  = c_claimFromSABB   + d_claimFromSABB;
    
    return new BalanceSheetResult(
    eligibleAvgBalance, sarBalance,
    c_statReserve,  c_eligibleBalNet, c_vof, c_dpf, c_hssShare, c_totalNII, c_vatCost, c_claimFromSABB,
    d_statReserve,  d_eligibleBalNet, d_vof, d_dpf, d_hssShare, d_totalNII, d_vatCost, d_claimFromSABB,
    f_sarBalance,   f_eligibleBalNet,  f_vof, f_dpf, f_hssShare, f_totalNII, f_vatCost, f_claimFromSABB
    );
    }

  static String pct(double rate) {
  // Trim trailing zeros but keep at least one decimal
  String s = String.format(”%.4f”, rate * 100);
  s = s.replaceAll(“0+$”, “”).replaceAll(”\.$”, “.0”);
  return s + “%”;
  }
