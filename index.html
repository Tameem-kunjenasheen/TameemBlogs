package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * This method calculates ALIMNA table using SUMIF logic
     *
     * @param data           Source data (filtered records from Step 1)
     * @param alimnaList     List of ALIMNA customer numbers
     * @return Map containing customer-wise balance and total
     */
    public static Map process(
            List> data,
            List alimnaList) {

        System.out.println("STEP: ALIMNA Table Calculation Started");

        // Final result map (maintains order)
        Map result = new LinkedHashMap<>();

        double total = 0;

        // Loop each ALIMNA customer (like 17159, 17332)
        for (String customerKey : alimnaList) {

            double sum = 0;

            // Loop all source data rows
            for (Map row : data) {

                // Get customer number from row
                String customer = row.get("CUSTOMER");

                // SUMIF condition (Excel logic)
                if (customerKey.equals(customer)) {

                    // Get balance and add
                    double balance = Double.parseDouble(row.get("BALANCE"));
                    sum += balance;
                }
            }

            // Store customer-wise result
            result.put(customerKey, sum);

            // Add to total
            total += sum;

            System.out.println("ALIMNA Customer: " + customerKey +
                    " → Balance: " + sum);
        }

        // Add total row
        result.put("TOTAL", total);

        System.out.println("ALIMNA TOTAL: " + total);

        return result;
    }
}
