package main;

import java.math.BigDecimal;
import java.util.*;

public class FullCategoryService {

    public static Map> process(
            List> sourceData,
            Map categoryMap) {

        System.out.println("===== ALL CATEGORY TABLES STARTED =====");

        // Final result → 5 tables
        Map> finalResult = new LinkedHashMap<>();

        // Step 1: Initialize categories
        List categories = Arrays.asList(
                "TRUST",
                "DCC",
                "GC",
                "PRIVATE FUND",
                "OTHER / INTERNAL"
        );

        for (String cat : categories) {
            finalResult.put(cat, new LinkedHashMap<>());
        }

        // Step 2: Loop category map
        for (Map.Entry entry : categoryMap.entrySet()) {

            String customer = entry.getKey();
            String category = entry.getValue().toUpperCase();

            // Normalize category name
            String finalCategory = getCategoryName(category);

            if (finalCategory == null) continue;

            // Calculate SUMIF
            BigDecimal sum = calculateSum(sourceData, customer);

            finalResult.get(finalCategory).put(customer, sum);

            System.out.println(finalCategory + " → " + customer + " : " + sum);
        }

        // Step 3: Calculate TOTAL for each category
        for (String cat : finalResult.keySet()) {

            BigDecimal total = BigDecimal.ZERO;

            for (BigDecimal val : finalResult.get(cat).values()) {
                total = total.add(val);
            }

            finalResult.get(cat).put("TOTAL", total);

            System.out.println(cat + " TOTAL: " + total);
        }

        System.out.println("===== ALL CATEGORY TABLES COMPLETED =====");

        return finalResult;
    }

    /**
     * Normalize category names
     */
    private static String getCategoryName(String category) {

        if (category.contains("TRUST")) return "TRUST";
        if (category.contains("DCC")) return "DCC";
        if (category.contains("GC")) return "GC";
        if (category.contains("PRIVATE")) return "PRIVATE FUND";
        if (category.contains("OTHER")) return "OTHER / INTERNAL";

        return null;
    }

    /**
     * SUMIF logic
     */
    private static BigDecimal calculateSum(
            List> sourceData,
            String customerKey) {

        BigDecimal sum = BigDecimal.ZERO;

        for (Map row : sourceData) {

            try {
                String customer = row.get("ALDCS");

                if (customerKey.equals(customer)) {

                    String valueStr = row.get("ALMVC1");

                    if (valueStr != null && !valueStr.isEmpty()) {

                        BigDecimal value = new BigDecimal(valueStr);

                        // divide by 100
                        BigDecimal balance = value.divide(new BigDecimal("100"));

                        sum = sum.add(balance);
                    }
                }

            } catch (Exception e) {
                System.out.println("Error row: " + row);
            }
        }

        return sum;
    }
}

package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * Main method to process ALIMNA & PIF calculations
     */
    public static Map process(
            List> sourceData,
            Map exceptionSet) {

        System.out.println("===== ALIMNA TABLE STARTED =====");

        // Result maps
        Map alimnaMap = new LinkedHashMap<>();
        Map pifMap = new LinkedHashMap<>();

        double alimnaTotal = 0;
        double pifTotal = 0;

        // Loop each exception customer
        for (Map.Entry entry : exceptionSet.entrySet()) {

            String customer = entry.getKey();
            String name = entry.getValue().toUpperCase();

            // Calculate SUMIF logic
            double sum = calculateSum(sourceData, customer);

            // Split ALIMNA / PIF
            if (name.contains("ALINMA")) {

                alimnaMap.put(customer, sum);
                alimnaTotal += sum;

                System.out.println("ALIMNA → " + customer + " : " + sum);

            } else if (name.contains("PUBLIC INVESTMENT")) {

                pifMap.put(customer, sum);
                pifTotal += sum;

                System.out.println("PIF → " + customer + " : " + sum);
            }
        }

        // Add totals
        alimnaMap.put("TOTAL", alimnaTotal);
        pifMap.put("TOTAL", pifTotal);

        // Final result
        Map result = new HashMap<>();
        result.put("ALIMNA", alimnaMap);
        result.put("PIF", pifMap);

        System.out.println("ALIMNA TOTAL: " + alimnaTotal);
        System.out.println("PIF TOTAL: " + pifTotal);

        System.out.println("===== ALIMNA TABLE COMPLETED =====");

        return result;
    }

    /**
     * SUMIF logic:
     * ALDCS = customer
     * ALMVC1 / 100 = balance
     */
    private static double calculateSum(
            List> sourceData,
            String customerKey) {

        double sum = 0;

        for (Map row : sourceData) {

            try {
                String customer = row.get("ALDCS");

                // Match customer
                if (customerKey.equals(customer)) {

                    // Get ALMVC1 and convert to balance
                    String valueStr = row.get("ALMVC1");

                    if (valueStr != null && !valueStr.isEmpty()) {

                        double value = Double.parseDouble(valueStr);
                        double balance = value / 100;

                        sum += balance;
                    }
                }

            } catch (Exception e) {
                System.out.println("Error processing row: " + row);
            }
        }

        return sum;
    }
}

package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * This method:
     * 1. Splits exceptionSet into ALIMNA & PIF
     * 2. Calculates SUMIF (customer-wise balance)
     * 3. Returns result
     */
    public static Map process(
            List> sourceData,
            Map exceptionSet) {

        System.out.println("===== ALIMNA TABLE PROCESS STARTED =====");

        // Result maps
        Map alimnaResult = new LinkedHashMap<>();
        Map pifResult = new LinkedHashMap<>();

        double alimnaTotal = 0;
        double pifTotal = 0;

        // Step 1: Split ALIMNA & PIF
        for (Map.Entry entry : exceptionSet.entrySet()) {

            String customer = entry.getKey();
            String name = entry.getValue().toUpperCase();

            // ALIMNA customers
            if (name.contains("ALINMA")) {

                double sum = calculateSum(sourceData, customer);
                alimnaResult.put(customer, sum);
                alimnaTotal += sum;

                System.out.println("ALIMNA → " + customer + " : " + sum);
            }

            // PIF customers
            else if (name.contains("PUBLIC INVESTMENT")) {

                double sum = calculateSum(sourceData, customer);
                pifResult.put(customer, sum);
                pifTotal += sum;

                System.out.println("PIF → " + customer + " : " + sum);
            }
        }

        // Add totals
        alimnaResult.put("TOTAL", alimnaTotal);
        pifResult.put("TOTAL", pifTotal);

        // Final output
        Map finalResult = new HashMap<>();
        finalResult.put("ALIMNA", alimnaResult);
        finalResult.put("PIF", pifResult);

        System.out.println("ALIMNA TOTAL: " + alimnaTotal);
        System.out.println("PIF TOTAL: " + pifTotal);

        System.out.println("===== ALIMNA TABLE PROCESS COMPLETED =====");

        return finalResult;
    }

    /**
     * SUMIF logic
     */
    private static double calculateSum(
            List> data,
            String customerKey) {

        double sum = 0;

        for (Map row : data) {

            String customer = row.get("CUSTOMER");

            if (customerKey.equals(customer)) {

                double balance = Double.parseDouble(row.get("BALANCE"));
                sum += balance;
            }
        }

        return sum;
    }
}
