  public class Jawaad17 {
  
  // =========================================================
  //  CONFIGURABLE VARIABLES — Edit these values as needed

  // Core balance inputs (SAR)
  static double L863 = 5_000_000.00;   // Eligible Average Balances reference
  static double C885 = 4_800_000.00;   // Eligible SAR Balance
  static double C888 = 336_000.00;     // Stat reserve deduction (C886 * 0.07)
  static double C889 = 4_464_000.00;   // Eligible balances (C886 - C888)
  
  // Percentages — hard-coded values replaced with named constants
  static double STAT_RESERVE_RATE      = 0.07;          // 7%   stat reserve deduction
  static double VOF_ANNUAL_RATE        = 3.75 / 100.0;  // 3.75% VOF annual rate
  static double DPF_ANNUAL_RATE        = 0.05 / 100.0;  // 0.05% Deposit Protection Fund
  static double VAT_RATE               = 0.15;          // 15%  VAT
  static double PIF_SPECIAL_RATE       = 0.15 / 100.0;  // 0.15%pa PIF Special Rate
  static int    DAYS_IN_PERIOD         = 28;            // number of days in period
  static int    DAYS_IN_YEAR           = 365;           // days in a year
  
  // Alinma Exemption adjustment (NIL NII)
  static double E877 = -120_000.00;    // Alinma Exemption (negative = deduction)
  static double D885 = C885;           // mirrors C885 after exemption column
  
  // =========================================================
  //  DERIVED / CALCULATED VARIABLES
  // =========================================================
  
  // — Before Exceptions column (C) —
  static double C886 = C885;                                        // Eligible SAR Balance
  static double C887 = C886 * STAT_RESERVE_RATE;                   // Less: Stat reserve
  // C888 = C886 - C887  (eligible balances after reserve)
  static double eligibleBalC = C886 - C887;
  
  // VOF / Deposit Protection Fund / HSS (Before Exceptions)
  static double C891 = (C889 * VOF_ANNUAL_RATE / DAYS_IN_YEAR) * DAYS_IN_PERIOD;   // Computed VOF VOF
  static double C892 = (C885 * DPF_ANNUAL_RATE / DAYS_IN_YEAR) * DAYS_IN_PERIOD;  // Deposit Protection Fund
  static double C893 = C891 - C892;                                                 // HSS share (C891 - C892)
  
  // Total NII to be booked (Before Exceptions)
  static double C894 = C893 * 0.88;    // Total NII * 0.88
  static double C895 = C893;           // base for VAT calc
  static double C896 = C895 * VAT_RATE; // VAT cost (15%)
  static double claimFromSABB_C = sumRange(C894, C895, C896); // =SUM(C894:C896) — Claim from SABB
  
  // — Alinma Exemption column (D) —
  static double D886 = D885;
  static double D887 = D886 * STAT_RESERVE_RATE;
  static double eligibleBalD = D886 - D887;
  
  static double D889 = eligibleBalD;
  static double D891 = (D889 * VOF_ANNUAL_RATE / DAYS_IN_YEAR) * DAYS_IN_PERIOD;
  static double D892 = (D885 * DPF_ANNUAL_RATE / DAYS_IN_YEAR) * DAYS_IN_PERIOD;
  static double D893 = D891 - D892;
  
  static double D894 = D893 * 0.88;
  static double D895 = D893;
  static double D896 = D895 * VAT_RATE;
  static double claimFromSABB_D = sumRange(D894, D895, D896);
  
  // — After Exceptions column (F) = SUM across columns —
  static double F885 = sumRange(C885, D885);
  static double F886 = sumRange(C886, D886);
  static double F888 = sumRange(eligibleBalC, eligibleBalD);
  static double F889 = sumRange(C889, D889);
  static double F891 = sumRange(C891, D891);
  static double F892 = sumRange(C892, D892);
  static double F893 = sumRange(C893, D893);
  static double F894 = sumRange(C894, D894);
  static double F895 = sumRange(C895, D895);
  static double F896 = sumRange(C896, D896);
  static double claimFromSABB_F = sumRange(claimFromSABB_C, claimFromSABB_D);
  
  // =========================================================
  //  HELPER
  // =========================================================
  static double sumRange(double… values) {
  double total = 0;
  for (double v : values) total += v;
  return total;
  }
  
  // =========================================================
  //  FORMATTING HELPERS
  // =========================================================
  static String fmt(double v) {
  return String.format(”%,15.2f”, v);
  }
  
  static String col(String text, int width) {
  if (text.length() > width) text = text.substring(0, width - 1);
  return String.format(”%-” + width + “s”, text);
  }
  
  static String colR(String text, int width) {
  if (text.length() > width) text = text.substring(0, width - 1);
  return String.format(”%” + width + “s”, text);
  }

  static void printRow(String sep, int w0, int w1, int w2, int w3, int w4, int w5,
  String desc, String c0, String c1, String c2, String c3, String c4) {
  System.out.printf(”| %-” + w0 + “s | %” + w1 + “s | %” + w2 + “s | %” + w3 + “s | %” + w4 + “s | %” + w5 + “s |%n”,
  desc, c0.trim(), c1.trim(), c2.trim(), c3.trim(), c4.trim());
  }
