import org.apache.poi.ss.usermodel.*;
import java.io.File;
import java.util.*;

public class ExcelReaderService {

    // Reads source Excel and applies filtering logic
    public List readSource(String filePath) throws Exception {

        List records = new ArrayList<>();

        Workbook wb = WorkbookFactory.create(new File(filePath));
        Sheet sheet = wb.getSheetAt(0);

        for (Row row : sheet) {

            // Skip header row
            if (row.getRowNum() == 0) continue;

            // Read required columns
            String bank = row.getCell(0).toString();     // ALDCG
            String aldcs = row.getCell(2).toString();    // ALDCS
            String product = row.getCell(8).toString();  // ALPRIE
            String currency = row.getCell(9).toString(); // ALCYCD

            // Balance in halalas
            double almvc1 = row.getCell(10).getNumericCellValue();

            // Apply filter conditions
            if (bank.equals("SABB") &&
                product.equals("CUT") &&
                currency.equals("SAR") &&
                almvc1 > 0) {

                // Convert halalas to SAR
                double balance = almvc1 / 100;

                // Add record to list
                records.add(new Record(aldcs, balance));
            }
        }

        wb.close();
        return records;
    }
}

import org.apache.poi.ss.usermodel.*;
import java.io.File;
import java.util.*;

public class MappingService {

    // Loads mapping file into memory
    public Map loadMapping(String filePath) throws Exception {

        Map map = new HashMap<>();

        Workbook wb = WorkbookFactory.create(new File(filePath));
        Sheet sheet = wb.getSheetAt(0);

        for (Row row : sheet) {

            if (row.getRowNum() == 0) continue;

            String aldcs = row.getCell(0).toString();
            String category = row.getCell(1).toString();
            String include = row.getCell(2).toString();

            map.put(aldcs, new Mapping(category, include));
        }

        wb.close();
        return map;
    }
}

📄 CalculationService.java
import java.util.*;

public class CalculationService {

    public Result calculate(List records,
                            Map mapping,
                            double rate,
                            int days) {

        Map totals = new HashMap<>();

        // Loop through each record
        for (Record r : records) {

            Mapping m = mapping.get(r.aldcs);

            // Skip if no mapping found
            if (m == null) continue;

            // Skip if excluded
            if (!m.include.equalsIgnoreCase("Y")) continue;

            // Add balance to category total
            totals.put(
                m.category,
                totals.getOrDefault(m.category, 0.0) + r.balance
            );
        }

        Result res = new Result();

        // Extract category totals
        res.trust = totals.getOrDefault("TRUST", 0.0);
        res.internal = totals.getOrDefault("INTERNAL", 0.0);
        res.privateFunds = totals.getOrDefault("PRIVATE", 0.0);

        // Total = sum of all categories
        res.total = totals.values()
                          .stream()
                          .mapToDouble(Double::doubleValue)
                          .sum();

        // Business logic
        res.dcc = res.total - res.privateFunds;
        res.gc = res.privateFunds;

        // NII calculation
        res.nii = (res.total * rate * days) / 360;

        // VAT
        res.vat = res.nii * 0.15;

        // Claim
        res.claim = res.nii + res.vat;

        return res;
    }
}

import org.apache.poi.xssf.usermodel.*;

public class ReportService {

    public String generate(Result r) throws Exception {

        String fileName = "output.xlsx";

        XSSFWorkbook wb = new XSSFWorkbook();
        XSSFSheet sheet = wb.createSheet("Summary");

        int row = 0;

        // Write each value clearly
        sheet.createRow(row).createCell(0).setCellValue("Total");
        sheet.getRow(row++).createCell(1).setCellValue(r.total);

        sheet.createRow(row).createCell(0).setCellValue("Trust");
        sheet.getRow(row++).createCell(1).setCellValue(r.trust);

        sheet.createRow(row).createCell(0).setCellValue("Internal");
        sheet.getRow(row++).createCell(1).setCellValue(r.internal);

import java.io.File;
import java.nio.file.*;
import java.text.SimpleDateFormat;
import java.util.Date;

public class BackupService {

    public void backup(String input, String output) throws Exception {

        // Create timestamp
        String time = new SimpleDateFormat("yyyyMMdd_HHmmss")
                        .format(new Date());

        // Create backup folder if not exists
        File dir = new File("bkp");
        if (!dir.exists()) dir.mkdir();

        // Copy input file
        Files.copy(
            Paths.get(input),
            Paths.get("bkp/input_" + time + ".xlsx"),
            StandardCopyOption.REPLACE_EXISTING
        );

        // Copy output file
        Files.copy(
            Paths.get(output),
            Paths.get("bkp/output_" + time + ".xlsx"),
            StandardCopyOption.REPLACE_EXISTING
        );
    }
}

public class NiiController {

    private ExcelReaderService reader = new ExcelReaderService();
    private MappingService mappingService = new MappingService();
    private CalculationService calc = new CalculationService();
    private ReportService report = new ReportService();
    private BackupService backup = new BackupService();

    public void execute(String sourceFile, String mappingFile,
                        double rate, int days) throws Exception {

        var records = reader.readSource(sourceFile);

        var mapping = mappingService.loadMapping(mappingFile);

        var result = calc.calculate(records, mapping, rate, days);

        String output = report.generate(result);

        backup.backup(sourceFile, output);
    }
}
