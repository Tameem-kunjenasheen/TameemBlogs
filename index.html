import java.util.*;

public class SourceDataProcessor {

    public static Map process(List> sourceData) {

        List> filtered = new ArrayList<>();

        // 1. Filter ALPRIE = CUT and ALDCG = SABB
        for (Map row : sourceData) {

            String product = row.get("ALPRIE");
            String bank = row.get("ALDCG");

            if ("CUT".equalsIgnoreCase(product) && "SABB".equalsIgnoreCase(bank)) {
                filtered.add(row);
            }
        }

        // 2. Remove zero balance
        filtered.removeIf(row -> {
            String val = row.get("ALMVC1");
            return val == null || val.trim().isEmpty() || Double.parseDouble(val) == 0;
        });

        // 3. Sort by ALMVC1 (ascending)
        filtered.sort(Comparator.comparingDouble(row ->
                Double.parseDouble(row.get("ALMVC1"))
        ));

        // 4. Create result map (CustomerNo → Balance)
        Map result = new LinkedHashMap<>();

        for (Map row : filtered) {

            String customerNo = row.get("ALACCS"); // adjust if different column
            double almvc1 = Double.parseDouble(row.get("ALMVC1"));

            double balance = almvc1 / 100.0;

            result.put(customerNo, balance);
        }

        return result;
    }
}

✅ 1. Common Excel Reader Utility (Reusable)

This is similar to your IpoDataExcelReader, but generic.

import org.apache.poi.ss.usermodel.*;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

import java.io.File;
import java.io.FileInputStream;
import java.util.*;

public class ExcelReaderUtil {

    public static List> readExcel(File file) throws Exception {

        List> rows = new ArrayList<>();

        try (FileInputStream fis = new FileInputStream(file);
             Workbook wb = new XSSFWorkbook(fis)) {

            Sheet sheet = wb.getSheetAt(0);

            Row headerRow = sheet.getRow(0);
            if (headerRow == null) {
                throw new RuntimeException("Excel file missing header row");
            }

            DataFormatter formatter = new DataFormatter();

            // Read headers
            List headers = new ArrayList<>();
            int lastCol = headerRow.getLastCellNum();

            for (int i = 0; i < lastCol; i++) {
                Cell cell = headerRow.getCell(i, Row.MissingCellPolicy.CREATE_NULL_AS_BLANK);
                String header = formatter.formatCellValue(cell).trim();

                if (header.isEmpty()) break;
                headers.add(header);
            }

            // Read data rows
            for (int i = 1; i <= sheet.getLastRowNum(); i++) {

                Row row = sheet.getRow(i);
                if (row == null) continue;

                Map rowMap = new LinkedHashMap<>();

                for (int j = 0; j < headers.size(); j++) {
                    Cell cell = row.getCell(j, Row.MissingCellPolicy.CREATE_NULL_AS_BLANK);
                    String value = formatter.formatCellValue(cell).trim();

                    rowMap.put(headers.get(j), value);
                }

                rows.add(rowMap);
            }
        }

        return rows;
    }
}

✅ 2. Loader Class for All 4 Files

Each file → separate map (as per your requirement)

import java.io.File;
import java.util.*;

public class DataLoader {

    // 1. Source Data
    public static List> loadSource(File file) throws Exception {
        return ExcelReaderUtil.readExcel(file);
    }

    // 2. Percentage Data
    public static Map> loadPercentage(File file) throws Exception {

        List> rows = ExcelReaderUtil.readExcel(file);
        Map> result = new HashMap<>();

        for (Map row : rows) {
            String key = row.get("Date"); // or unique column
            result.put(key, row);
        }

        return result;
    }

    // 3. Category Data
    public static Map loadCategory(File file) throws Exception {

        List> rows = ExcelReaderUtil.readExcel(file);
        Map result = new HashMap<>();

        for (Map row : rows) {
            String customerNo = row.get("Customer No");
            String category = row.get("Category"); // adjust column name

            result.put(customerNo, category);
        }

        return result;
    }

    // 4. Alinma Exception Data
    public static Set loadException(File file) throws Exception {

        List> rows = ExcelReaderUtil.readExcel(file);
        Set exceptionSet = new HashSet<>();

        for (Map row : rows) {
            String customerNo = row.get("Customer No");
            exceptionSet.add(customerNo);
        }

        return exceptionSet;
    }
}
