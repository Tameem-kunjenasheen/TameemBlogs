    public BalanceSheetResult calculateBalanceSheet(

            double eligibleAvgBalance,
            double sarBalance,
            double statReserveRate,
            double vofRate,
            double dpfRate,
            double niiFactor,
            double vatRate,
            int days,
            int yearDays

    ) {

        BalanceSheetResult r = new BalanceSheetResult();

        // =====================================================
        // BEFORE EXCEPTION
        // =====================================================

        r.cEligibleAvgBalance = eligibleAvgBalance;
        r.cSarBalance = sarBalance;

        r.cStatReserve =
                sarBalance * statReserveRate;

        r.cEligibleBalNet =
                sarBalance - r.cStatReserve;

        r.cVof =
                (r.cEligibleBalNet * vofRate / yearDays)
                        * days;

        r.cDpf =
                (sarBalance * dpfRate / yearDays)
                        * days;

        r.cHssShare =
                r.cVof - r.cDpf;

        r.cTotalNii =
                r.cHssShare * niiFactor;

        r.cVatCost =
                r.cTotalNii * vatRate;

        r.cClaimFromSabb =
                r.cTotalNii + r.cVatCost;

        // =====================================================
        // ALINMA EXCEPTION
        // =====================================================

        r.dEligibleAvgBalance = 100000;
        r.dSarBalance = 100000;

        r.dStatReserve =
                r.dSarBalance * statReserveRate;

        r.dEligibleBalNet =
                r.dSarBalance - r.dStatReserve;

        r.dVof =
                (r.dEligibleBalNet * vofRate / yearDays)
                        * days;

        r.dDpf =
                (r.dSarBalance * dpfRate / yearDays)
                        * days;

        r.dHssShare =
                r.dVof - r.dDpf;

        r.dTotalNii =
                r.dHssShare * niiFactor;

        r.dVatCost =
                r.dTotalNii * vatRate;

        r.dClaimFromSabb =
                r.dTotalNii + r.dVatCost;

        // =====================================================
        // PIF EXCEPTION
        // =====================================================

        r.eEligibleAvgBalance = 50000;
        r.eSarBalance = 50000;

        r.eStatReserve =
                r.eSarBalance * statReserveRate;

        r.eEligibleBalNet =
                r.eSarBalance - r.eStatReserve;

        r.eVof =
                (r.eEligibleBalNet * vofRate / yearDays)
                        * days;

        r.eDpf =
                (r.eSarBalance * dpfRate / yearDays)
                        * days;

        r.eHssShare =
                r.eVof - r.eDpf;

        r.eTotalNii =
                r.eHssShare * niiFactor;

        r.eVatCost =
                r.eTotalNii * vatRate;

        r.eClaimFromSabb =
                r.eTotalNii + r.eVatCost;

        // =====================================================
        // PIF SPECIAL RATE
        // =====================================================

        r.fEligibleAvgBalance = 25000;
        r.fSarBalance = 25000;

        r.fStatReserve =
                r.fSarBalance * statReserveRate;

        r.fEligibleBalNet =
                r.fSarBalance - r.fStatReserve;

        r.fVof =
                (r.fEligibleBalNet * vofRate / yearDays)
                        * days;

        r.fDpf =
                (r.fSarBalance * dpfRate / yearDays)
                        * days;

        r.fHssShare =
                r.fVof - r.fDpf;

        r.fTotalNii =
                r.fHssShare * niiFactor;

        r.fVatCost =
                r.fTotalNii * vatRate;

        r.fClaimFromSabb =
                r.fTotalNii + r.fVatCost;

        // =====================================================
        // AFTER EXCEPTION
        // SUM OF ALL COLUMNS
        // =====================================================

        r.gEligibleAvgBalance =
                r.cEligibleAvgBalance
                        + r.dEligibleAvgBalance
                        + r.eEligibleAvgBalance
                        + r.fEligibleAvgBalance;

        r.gSarBalance =
                r.cSarBalance
                        + r.dSarBalance
                        + r.eSarBalance
                        + r.fSarBalance;

        r.gStatReserve =
                r.cStatReserve
                        + r.dStatReserve
                        + r.eStatReserve
                        + r.fStatReserve;

        r.gEligibleBalNet =
                r.cEligibleBalNet
                        + r.dEligibleBalNet
                        + r.eEligibleBalNet
                        + r.fEligibleBalNet;

        r.gVof =
                r.cVof
                        + r.dVof
                        + r.eVof
                        + r.fVof;

        r.gDpf =
                r.cDpf
                        + r.dDpf
                        + r.eDpf
                        + r.fDpf;

        r.gHssShare =
                r.cHssShare
                        + r.dHssShare
                        + r.eHssShare
                        + r.fHssShare;

        r.gTotalNii =
                r.cTotalNii
                        + r.dTotalNii
                        + r.eTotalNii
                        + r.fTotalNii;

        r.gVatCost =
                r.cVatCost

        r.gClaimFromSabb =

                        + r.fClaimFromSabb;

        return r;
    }

    /**
     * Main reusable calculation method
     *
     * You can call this method from another class
     */
    public BalanceSheetResult calculateBalanceSheet(
            double eligibleAvgBalance,
            double sarBalance,
            double statReserveRate,
            double vofAnnualRate,
            double dpfAnnualRate,
            double niiShareRate,
            double vatRate,
            int daysInPeriod,
            int daysInYear) {
        // Create result object
        BalanceSheetResult result = new BalanceSheetResult();
        // ============================
        // STORE INPUT VALUES
        // ============================
        result.setEligibleAvgBalance(eligibleAvgBalance);
        result.setSarBalance(sarBalance);
        // ============================
        // BEFORE EXCEPTION (C)
        // ============================
        double cStatReserve = sarBalance * statReserveRate;
        double cEligibleNet = sarBalance - cStatReserve;
        double cVof =
                (cEligibleNet * vofAnnualRate / daysInYear)
                        * daysInPeriod;
        double cDpf =
                (sarBalance * dpfAnnualRate / daysInYear)
                        * daysInPeriod;
        double cHss = cVof - cDpf;
        double cNii = cHss * niiShareRate;
        double cVat = cHss * vatRate;
        double cClaim = cNii + cVat;
        // Store values
        result.setcStatReserve(cStatReserve);
        result.setcEligibleBalNet(cEligibleNet);
        result.setcVof(cVof);
        result.setcDpf(cDpf);
        result.setcHssShare(cHss);
        result.setcTotalNii(cNii);
        result.setcVatCost(cVat);
        result.setcClaimFromSabb(cClaim);
        // ============================
        // D COLUMN (ALINMA EXCEPTION)
        // ============================
        double dStatReserve = sarBalance * statReserveRate;
        double dEligibleNet = sarBalance - dStatReserve;
        double dVof =
                (dEligibleNet * vofAnnualRate / daysInYear)
                        * daysInPeriod;
        double dDpf =
                (sarBalance * dpfAnnualRate / daysInYear)
                        * daysInPeriod;
        double dHss = dVof - dDpf;
        double dNii = dHss * niiShareRate;
        double dVat = dHss * vatRate;
        double dClaim = dNii + dVat;
        // Store values
        result.setdStatReserve(dStatReserve);
        result.setdEligibleBalNet(dEligibleNet);
        result.setdVof(dVof);
        result.setdDpf(dDpf);
        result.setdHssShare(dHss);
        result.setdTotalNii(dNii);
        result.setdVatCost(dVat);
        result.setdClaimFromSabb(dClaim);
        // ============================
        // AFTER EXCEPTION (F)
        // ============================
        result.setfSarBalance(sarBalance + sarBalance);
        result.setfEligibleBalNet(
                cEligibleNet + dEligibleNet);
        result.setfVof(cVof + dVof);
        result.setfDpf(cDpf + dDpf);
        result.setfHssShare(cHss + dHss);
        result.setfTotalNii(cNii + dNii);
        result.setfVatCost(cVat + dVat);
        result.setfClaimFromSabb(cClaim + dClaim);
        // Return final result object
        return result;
