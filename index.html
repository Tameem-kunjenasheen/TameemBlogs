import org.apache.poi.ss.usermodel.*;
import java.io.File;
import java.util.*;

public class ExcelReaderService {

    public List readSource(String filePath) throws Exception {

        List records = new ArrayList<>();

        Workbook wb = WorkbookFactory.create(new File(filePath));
        Sheet sheet = wb.getSheetAt(0);

        for (Row row : sheet) {

            if (row.getRowNum() == 0) continue;

            String bank = row.getCell(0).toString();
            String aldcs = row.getCell(2).toString();
            String product = row.getCell(8).toString();
            String currency = row.getCell(9).toString();

            double almvc1 = row.getCell(10).getNumericCellValue();

            // Apply filter
            if (bank.equals("SABB") &&
                product.equals("CUT") &&
                currency.equals("SAR") &&
                almvc1 > 0) {

                double balance = almvc1 / 100;

                records.add(new Record(aldcs, balance));
            }
        }

        wb.close();
        return records;
    }
}
📄 4. CalculationService.java (FULL LOGIC 🔥)
import java.util.*;

public class CalculationService {

    public Map groupByCategory(List records) {

        Map map = new HashMap<>();

        for (Record r : records) {

            map.put(
                r.aldcs,
                map.getOrDefault(r.aldcs, 0.0) + r.balance
            );
        }

        return map;
    }

    public Map calculate(Map totals, InputParams p) {

        Map result = new HashMap<>();

        double totalBalance = totals.values().stream().mapToDouble(Double::doubleValue).sum();

        // Eligible balance
        double statReserve = totalBalance * p.statReserve;
        double eligibleBalance = totalBalance - statReserve;

        // VOF
        double vof = eligibleBalance * (p.vofRate / 100) * p.days / 365;

        // DPF
        double dpf = eligibleBalance * (p.dpfRate / 100) * p.days / 365;

        // HSS
        double hss = vof - dpf;

        // NII
        double nii = totalBalance * p.repoRate * p.days / 360;

        // Adjusted NII
        double niiAdjusted = nii * p.niiFactor;

        // VAT
        double vat = niiAdjusted * p.vatRate;

        // Claim
        double claim = niiAdjusted + vat;

        // Store all values
        result.put("TOTAL", totalBalance);
        result.put("STAT_RESERVE", statReserve);
        result.put("ELIGIBLE", eligibleBalance);
        result.put("VOF", vof);
        result.put("DPF", dpf);
        result.put("HSS", hss);
        result.put("NII", niiAdjusted);
        result.put("VAT", vat);
        result.put("CLAIM", claim);

        return result;
    }
}

import org.apache.poi.xssf.usermodel.*;
import java.io.FileOutputStream;
import java.util.Map;

public class ReportService {

    public String generate(Map r) throws Exception {

        XSSFWorkbook wb = new XSSFWorkbook();
        XSSFSheet sheet = wb.createSheet("Working");

        int row = 0;

        // Title
        sheet.createRow(row++).createCell(0).setCellValue("NII ESTIMATION");

        row++;

        // Eligible Balance Section
        sheet.createRow(row).createCell(0).setCellValue("Total Balance");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("TOTAL"));

        sheet.createRow(row).createCell(0).setCellValue("Stat Reserve");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("STAT_RESERVE"));

        sheet.createRow(row).createCell(0).setCellValue("Eligible Balance");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("ELIGIBLE"));

        sheet.createRow(row).createCell(0).setCellValue("VOF");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("VOF"));

        sheet.createRow(row).createCell(0).setCellValue("DPF");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("DPF"));

        sheet.createRow(row).createCell(0).setCellValue("HSS Share");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("HSS"));

        sheet.createRow(row).createCell(0).setCellValue("NII");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("NII"));

        sheet.createRow(row).createCell(0).setCellValue("VAT");

        wb.close();

        return "Output.xlsx";
    }

import java.io.File;
import java.nio.file.*;
import java.text.SimpleDateFormat;
import java.util.Date;

public class BackupService {

    public void backup(String input, String output) throws Exception {

        String time = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());

        File dir = new File("bkp");
        if (!dir.exists()) dir.mkdir();

        Files.copy(Paths.get(input), Paths.get("bkp/input_" + time + ".xlsx"));
        Files.copy(Paths.get(output), Paths.get("bkp/output_" + time + ".xlsx"));
    }
}
📄 7. NiiController.java
import java.util.*;

public class NiiController {

    ExcelReaderService reader = new ExcelReaderService();
    CalculationService calc = new CalculationService();
    ReportService report = new ReportService();
    BackupService backup = new BackupService();

    public void execute(String sourceFile, InputParams params) throws Exception {

        List records = reader.readSource(sourceFile);

        Map grouped = calc.groupByCategory(records);

        Map result = calc.calculate(grouped, params);

        String output = report.generate(result);

        backup.backup(sourceFile, output);
    }
}
