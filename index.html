✅ 2. SourceFilterService.java (STEP 1)

package service;

import model.Record;
import java.util.*;

/**
 * Step 1:
 * Filter + calculate balance
 */
public class SourceFilterService {

    public double process(List records, List filtered) {

        double totalSum = 0; // final sum of all valid balances

        for (Record r : records) {

            // Apply filters:
            // Only CUT, SAR, SABB records allowed
            if ("CUT".equalsIgnoreCase(r.getHalal()) &&
                "SAR".equalsIgnoreCase(r.getCurrency()) &&
                "SABB".equalsIgnoreCase(r.getBank())) {

                // Add to filtered list
                filtered.add(r);

                // Add to total
                totalSum += r.getBalance();
            }
        }

        return totalSum;
    }
}

✅ 3. Table1Service.java (STEP 2)

package service;

import model.Record;
import java.util.*;

/**
 * Step 2:
 * Alimah + PIF calculation (SUMIF logic)
 */
public class Table1Service {

    public Map process(List data,
                                       List alimahCustomers,
                                       List pifCustomers) {

        Map result = new HashMap<>();

        double alimahTotal = 0;
        double pifTotal = 0;

        for (Record r : data) {

            // Alimah SUMIF
            if (alimahCustomers.contains(r.getCustomer())) {
                alimahTotal += r.getBalance();
            }

            // PIF SUMIF
            if (pifCustomers.contains(r.getCustomer())) {
                pifTotal += r.getBalance();
            }
        }

        result.put("ALIMAH_TOTAL", alimahTotal);
        result.put("PIF_TOTAL", pifTotal);

        return result;
    }
}

✅ 4. Table2Service.java (STEP 3)

package service;

/**
 * Step 3:
 * Financial calculations
 */
public class Table2Service {

    public double calculateNII(double totalBalance,
                               double alimahTotal,
                               double statReserve,
                               double vofRate,
                               double dpfRate,
                               double niiFactor) {

        // Eligible balance after removing alimah
        double eligible = totalBalance - alimahTotal;

        // Stat reserve deduction
        double stat = eligible * statReserve;

        // After deduction
        double net = eligible - stat;

        // VOF calculation
        double vof = net * vofRate;

        // Deposit Protection Fund
        double dpf = net * dpfRate;

        // HSS share
        double hss = vof - dpf;

        // Final NII
        double nii = hss * niiFactor;

        return nii;
    }
}

package service;

import model.Record;
import java.util.*;

/**
 * Step 4:
 * Group by category
 */
public class CategoryService {

    public Map process(List data) {

        Map categorySum = new HashMap<>();

        for (Record r : data) {

            String cat = r.getCategory();

            // Initialize if not exists
            if (!categorySum.containsKey(cat)) {
                categorySum.put(cat, 0.0);
            }

            // Add balance
            categorySum.put(cat, categorySum.get(cat) + r.getBalance());
        }

        return categorySum;
    }
}

package service;

import java.util.*;

/**
 * Step 5:
 * DCC + GC calculation
 */
public class Table3Service {

    public Map process(Map categorySum,
                                       double totalBalance) {

        Map result = new HashMap<>();

        double trust = categorySum.getOrDefault("TRUST", 0.0);
        double dcc = categorySum.getOrDefault("DCC", 0.0);
        double internal = categorySum.getOrDefault("INTERNAL", 0.0);
        double privateFund = categorySum.getOrDefault("PRIVATE", 0.0);

        // DCC total
        double dccTotal = trust + dcc + internal + privateFund;

        // GC/FS
        double gc = totalBalance - dccTotal;

        result.put("DCC", dccTotal);
        result.put("GC", gc);

        return result;
    }
}

package service;

import java.util.*;

/**
 * Step 6:
 * Voucher calculations
 */
public class VoucherService {

    public Map process(double totalBalance,
                                       double dcc,
                                       double gc,
                                       double nii,
                                       double vatRate) {

        Map result = new HashMap<>();

        // Base value
        double base = totalBalance - (dcc + gc);

        // NII estimate
        double niiEstimate = nii;

        // VAT
        double vat = niiEstimate * vatRate;

        result.put("BASE", base);
        result.put("NII", niiEstimate);
        result.put("VAT", vat);

        return result;
    }
}
