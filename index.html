package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * This method:
     * 1. Splits exceptionSet into ALIMNA & PIF
     * 2. Calculates SUMIF (customer-wise balance)
     * 3. Returns result
     */
    public static Map process(
            List> sourceData,
            Map exceptionSet) {

        System.out.println("===== ALIMNA TABLE PROCESS STARTED =====");

        // Result maps
        Map alimnaResult = new LinkedHashMap<>();
        Map pifResult = new LinkedHashMap<>();

        double alimnaTotal = 0;
        double pifTotal = 0;

        // Step 1: Split ALIMNA & PIF
        for (Map.Entry entry : exceptionSet.entrySet()) {

            String customer = entry.getKey();
            String name = entry.getValue().toUpperCase();

            // ALIMNA customers
            if (name.contains("ALINMA")) {

                double sum = calculateSum(sourceData, customer);
                alimnaResult.put(customer, sum);
                alimnaTotal += sum;

                System.out.println("ALIMNA → " + customer + " : " + sum);
            }

            // PIF customers
            else if (name.contains("PUBLIC INVESTMENT")) {

                double sum = calculateSum(sourceData, customer);
                pifResult.put(customer, sum);
                pifTotal += sum;

                System.out.println("PIF → " + customer + " : " + sum);
            }
        }

        // Add totals
        alimnaResult.put("TOTAL", alimnaTotal);
        pifResult.put("TOTAL", pifTotal);

        // Final output
        Map finalResult = new HashMap<>();
        finalResult.put("ALIMNA", alimnaResult);
        finalResult.put("PIF", pifResult);

        System.out.println("ALIMNA TOTAL: " + alimnaTotal);
        System.out.println("PIF TOTAL: " + pifTotal);

        System.out.println("===== ALIMNA TABLE PROCESS COMPLETED =====");

        return finalResult;
    }

    /**
     * SUMIF logic
     */
    private static double calculateSum(
            List> data,
            String customerKey) {

        double sum = 0;

        for (Map row : data) {

            String customer = row.get("CUSTOMER");

            if (customerKey.equals(customer)) {

                double balance = Double.parseDouble(row.get("BALANCE"));
                sum += balance;
            }
        }

        return sum;
    }
}

package main;

import java.util.*;

public class MainApp {

    public static void main(String[] args) {

        try {

            System.out.println("===== APPLICATION STARTED =====");

            // ---------------------------
            // STEP 1: MOCK SOURCE DATA
            // ---------------------------
            List> sourceData = new ArrayList<>();

            sourceData.add(createRow("17159", "1000"));
            sourceData.add(createRow("17159", "2000"));
            sourceData.add(createRow("17332", "500"));
            sourceData.add(createRow("16482", "3000"));

            // ---------------------------
            // STEP 2: EXCEPTION SET
            // ---------------------------
            Map exceptionSet = new HashMap<>();

            exceptionSet.put("17332", "ALINMA NOMU MKT EQUITY FUND");
            exceptionSet.put("16482", "PUBLIC INVESTMENT FUND -EQTY");
            exceptionSet.put("17159", "ALINMA FREE-STYLE EQ FUND");

            System.out.println("Exception Data: " + exceptionSet);

            // ---------------------------
            // STEP 3: CALL SERVICE
            // ---------------------------
            Map result =
                    AlimnaTableService.process(sourceData, exceptionSet);

            // ---------------------------
            // STEP 4: PRINT RESULT
            // ---------------------------
            System.out.println("\n===== FINAL RESULT =====");

            Map alimna =
                    (Map) result.get("ALIMNA");

            Map pif =
                    (Map) result.get("PIF");

            System.out.println("ALIMNA TABLE: " + alimna);
            System.out.println("PIF TABLE: " + pif);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    // Helper method to create row
    private static Map createRow(String cust, String bal) {

        Map row = new HashMap<>();
        row.put("CUSTOMER", cust);
        row.put("BALANCE", bal);

        return row;
    }
}
