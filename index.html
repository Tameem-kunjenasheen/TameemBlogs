package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * This method:
     * 1. Splits exceptionSet into ALIMNA & PIF
     * 2. Calculates SUMIF (customer-wise balance)
     * 3. Returns result
     */
    public static Map process(
            List> sourceData,
            Map exceptionSet) {

        System.out.println("===== ALIMNA TABLE PROCESS STARTED =====");

        // Result maps
        Map alimnaResult = new LinkedHashMap<>();
        Map pifResult = new LinkedHashMap<>();

        double alimnaTotal = 0;
        double pifTotal = 0;

        // Step 1: Split ALIMNA & PIF
        for (Map.Entry entry : exceptionSet.entrySet()) {

            String customer = entry.getKey();
            String name = entry.getValue().toUpperCase();

            // ALIMNA customers
            if (name.contains("ALINMA")) {

                double sum = calculateSum(sourceData, customer);
                alimnaResult.put(customer, sum);
                alimnaTotal += sum;

                System.out.println("ALIMNA → " + customer + " : " + sum);
            }

            // PIF customers
            else if (name.contains("PUBLIC INVESTMENT")) {

                double sum = calculateSum(sourceData, customer);
                pifResult.put(customer, sum);
                pifTotal += sum;

                System.out.println("PIF → " + customer + " : " + sum);
            }
        }

        // Add totals
        alimnaResult.put("TOTAL", alimnaTotal);
        pifResult.put("TOTAL", pifTotal);

        // Final output
        Map finalResult = new HashMap<>();
        finalResult.put("ALIMNA", alimnaResult);
        finalResult.put("PIF", pifResult);

        System.out.println("ALIMNA TOTAL: " + alimnaTotal);
        System.out.println("PIF TOTAL: " + pifTotal);

        System.out.println("===== ALIMNA TABLE PROCESS COMPLETED =====");

        return finalResult;
    }

    /**
     * SUMIF logic
     */
    private static double calculateSum(
            List> data,
            String customerKey) {

        double sum = 0;

        for (Map row : data) {

            String customer = row.get("CUSTOMER");

            if (customerKey.equals(customer)) {

                double balance = Double.parseDouble(row.get("BALANCE"));
                sum += balance;
            }
        }

        return sum;
    }
}

package main;

import java.util.*;

public class AlimnaTableService {

    /**
     * This method calculates ALIMNA table using SUMIF logic
     *
     * @param data           Source data (filtered records from Step 1)
     * @param alimnaList     List of ALIMNA customer numbers
     * @return Map containing customer-wise balance and total
     */
    public static Map process(
            List> data,
            List alimnaList) {

        System.out.println("STEP: ALIMNA Table Calculation Started");

        // Final result map (maintains order)
        Map result = new LinkedHashMap<>();

        double total = 0;

        // Loop each ALIMNA customer (like 17159, 17332)
        for (String customerKey : alimnaList) {

            double sum = 0;

            // Loop all source data rows
            for (Map row : data) {

                // Get customer number from row
                String customer = row.get("CUSTOMER");

                // SUMIF condition (Excel logic)
                if (customerKey.equals(customer)) {

                    // Get balance and add
                    double balance = Double.parseDouble(row.get("BALANCE"));
                    sum += balance;
                }
            }

            // Store customer-wise result
            result.put(customerKey, sum);

            // Add to total
            total += sum;

            System.out.println("ALIMNA Customer: " + customerKey +
                    " → Balance: " + sum);
        }

        // Add total row
        result.put("TOTAL", total);

        System.out.println("ALIMNA TOTAL: " + total);

        return result;
    }
}
