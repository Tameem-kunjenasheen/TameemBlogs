package mainUI;

import java.io.FileInputStream;
import java.util.*;
import org.apache.poi.ss.usermodel.*;

public class SourceReaderService {

    public List read(String filePath) throws Exception {

        List records = new ArrayList<>();

        System.out.println("STEP 1: Reading & Filtering Source File");

        Workbook wb = WorkbookFactory.create(new FileInputStream(filePath));
        Sheet sheet = wb.getSheetAt(0);

        // Header mapping
        Row header = sheet.getRow(0);
        Map map = new HashMap<>();

        for (Cell c : header) {
            map.put(c.getStringCellValue().trim(), c.getColumnIndex());
        }

        for (int i = 1; i <= sheet.getLastRowNum(); i++) {

            Row r = sheet.getRow(i);
            if (r == null) continue;

            String bank = getString(r, map.get("ALDCG"));
            String product = getString(r, map.get("ALPRIE"));
            String currency = getString(r, map.get("ALCYCD"));
            String customer = getString(r, map.get("ALACCS"));

            double almvc1 = getNumeric(r, map.get("ALMVC1"));

            // Filters
            if (!"CUT".equalsIgnoreCase(product)) continue;
            if (!"SAR".equalsIgnoreCase(currency)) continue;
            if (!"SABB".equalsIgnoreCase(bank)) continue;

            double balance = almvc1 / 100.0;
            if (balance == 0) continue;

            Record rec = new Record();
            rec.setBank(bank);
            rec.setProduct(product);
            rec.setCurrency(currency);
            rec.setCustomer(customer);
            rec.setAlmvc1(almvc1);
            rec.setBalance(balance);

            records.add(rec);
        }

        System.out.println("Filtered Records: " + records.size());
        return records;
    }

    private String getString(Row r, Integer i) {
        if (i == null) return "";
        Cell c = r.getCell(i);
        return c == null ? "" : c.toString();
    }

    private double getNumeric(Row r, Integer i) {
        if (i == null) return 0;
        Cell c = r.getCell(i);
        if (c == null) return 0;
        return c.getCellType() == CellType.NUMERIC ? c.getNumericCellValue() : 0;
    }
}

import java.util.*;

public class Table2Service {

    public Map calculate(double totalBalance,
                                         double alimah,
                                         Map rates) {

        double stat = totalBalance * rates.get("statReserve");
        double eligible = totalBalance - stat;

        double vof = eligible * rates.get("vofRate") / 365 * rates.get("days");
        double dpf = eligible * rates.get("dpfRate") / 365 * rates.get("days");

        double hss = vof - dpf;

        double nii = hss * rates.get("niiFactor");
        double vat = nii * rates.get("vatRate");

        map.put("eligible", eligible);
        map.put("vof", vof);
        map.put("hss", hss);
        map.put("nii", nii);
        map.put("vat", vat);

        return map;

import model.Record;
import org.apache.poi.ss.usermodel.*;

import java.io.FileInputStream;
import java.util.*;

/**
 * STEP 1:
 * Read Excel → Filter → Convert → Sort → Return Records
 */
public class SourceReaderService {

    public List readAndFilter(String filePath) throws Exception {

        // List to store final filtered records
        List filteredRecords = new ArrayList<>();

        // Open Excel file
        FileInputStream fis = new FileInputStream(filePath);

        // Create workbook (POI 3.7 compatible)
        Workbook workbook = WorkbookFactory.create(fis);

        // Get first sheet (Source sheet)
        Sheet sheet = workbook.getSheetAt(0);

        System.out.println("STEP 1: Reading Source File...");
        System.out.println("Total Rows: " + sheet.getLastRowNum());

        // -------------------------------
        // STEP 1.1: Read Header Row
        // -------------------------------
        Row headerRow = sheet.getRow(0);

        // Map to store column name → index
        Map headerMap = new HashMap<>();

        // Loop through header cells
        for (Cell cell : headerRow) {

            // Get column name
            String columnName = cell.getStringCellValue().trim();

            // Store column index
            headerMap.put(columnName, cell.getColumnIndex());

            System.out.println("Mapped Column: " + columnName + " → Index: " + cell.getColumnIndex());
        }

        // -------------------------------
        // STEP 1.2: Read Data Rows
        // -------------------------------
        for (int i = 1; i <= sheet.getLastRowNum(); i++) {

            // Get row
            Row row = sheet.getRow(i);

            // Skip empty row
            if (row == null) continue;

            // Create new Record object
            Record record = new Record();

            // -------------------------------
            // Read each required column
            // -------------------------------

            // Customer (ALDCS)
            record.setCustomer(getStringValue(row, headerMap.get("ALDCS")));

            // Category (ALACX)
            record.setCategory(getStringValue(row, headerMap.get("ALACX")));

            // Product Type (ALPRIE → CUT)
            record.setProductType(getStringValue(row, headerMap.get("ALPRIE")));

            // Currency (ALCYCD → SAR)
            record.setCurrency(getStringValue(row, headerMap.get("ALCYCD")));

            // Bank (ALGMAB → SABB)
            record.setBank(getStringValue(row, headerMap.get("ALGMAB")));

            // -------------------------------
            // Read ALMVC1 and convert to balance
            // -------------------------------
            double almvc1 = getNumericValue(row, headerMap.get("ALMVC1"));

            // Convert halalas → SAR
            double balance = almvc1 / 100;

            record.setBalance(balance);

            // -------------------------------
            // FILTER CONDITIONS
            // -------------------------------
            if ("CUT".equalsIgnoreCase(record.getProductType()) &&
                "SAR".equalsIgnoreCase(record.getCurrency()) &&
                "SABB".equalsIgnoreCase(record.getBank())) {

                // Remove zero balance rows
                if (balance > 0) {

                    filteredRecords.add(record);

                    System.out.println("Accepted Row " + i +
                            " | Customer=" + record.getCustomer() +
                            " | Balance=" + balance);
                }
            }
        }

        // -------------------------------
        // STEP 1.3: SORT (ALMVC1 ascending)
        // -------------------------------
        Collections.sort(filteredRecords, new Comparator() {
            public int compare(Record r1, Record r2) {
                return Double.compare(r1.getBalance(), r2.getBalance());
            }
        });

        System.out.println("Filtered Records Count: " + filteredRecords.size());

        // Close file
        workbook.close();
        fis.close();

        return filteredRecords;
    }

    // -------------------------------
    // Utility Method: Safe String Read
    // -------------------------------
    private String getStringValue(Row row, Integer colIndex) {

        // Get cell
        Cell cell = row.getCell(colIndex);

        if (cell == null) return "";

        // Convert to string safely
        if (cell.getCellType() == Cell.CELL_TYPE_STRING) {
            return cell.getStringCellValue().trim();
        } else {
            return cell.toString().trim();
        }
    }

    // -------------------------------
    // Utility Method: Safe Numeric Read
    // -------------------------------
    private double getNumericValue(Row row, Integer colIndex) {

        // Get cell
        Cell cell = row.getCell(colIndex);

        if (cell == null) return 0;

        // If numeric
        if (cell.getCellType() == Cell.CELL_TYPE_NUMERIC) {
            return cell.getNumericCellValue();
        } else {
            try {
                return Double.parseDouble(cell.toString());
            } catch (Exception e) {
                return 0;
            }
        }
    }
}

package service;

/**
 * Step 3:
 * Financial calculations
 */
public class Table2Service {

    public double calculateNII(double totalBalance,
                               double alimahTotal,
                               double statReserve,
                               double vofRate,
                               double dpfRate,
                               double niiFactor) {

        // Eligible balance after removing alimah
        double eligible = totalBalance - alimahTotal;

        // Stat reserve deduction
        double stat = eligible * statReserve;

        // After deduction
        double net = eligible - stat;

        // VOF calculation
        double vof = net * vofRate;

        // Deposit Protection Fund
        double dpf = net * dpfRate;

        // HSS share
        double hss = vof - dpf;

        // Final NII
        double nii = hss * niiFactor;

        return nii;
    }
}
