import org.apache.poi.ss.usermodel.*;        // For Excel handling
import org.apache.poi.xssf.usermodel.XSSFWorkbook; // For .xlsx files
import java.io.FileInputStream;              // To read file
import java.util.*;                          // For List

public class ExcelReaderService {

    // Method to read source Excel file
    public List readSource(String filePath) throws Exception {

        // Create list to store records
        List records = new ArrayList();

        // Open file input stream
        FileInputStream fis = new FileInputStream(filePath);

        // Create workbook object
        Workbook wb = new XSSFWorkbook(fis);

        // Get first sheet
        Sheet sheet = wb.getSheetAt(0);

        // Loop through all rows
        for (Row row : sheet) {

            // Skip header row
            if (row.getRowNum() == 0) continue;

            // Read required cells
            Cell bankCell = row.getCell(0);     // ALDCG
            Cell aldcsCell = row.getCell(2);    // ALDCS
            Cell productCell = row.getCell(8);  // ALPRIE
            Cell currencyCell = row.getCell(9); // ALCYCD
            Cell balanceCell = row.getCell(10); // ALMVC1

            // Skip if important cells missing
            if (bankCell == null || balanceCell == null) continue;

            // Convert cells to string
            String bank = bankCell.toString();
            String aldcs = aldcsCell.toString();
            String product = productCell.toString();
            String currency = currencyCell.toString();

            double almvc1 = 0;

            // Handle numeric safely (POI 3.7)
            if (balanceCell.getCellType() == Cell.CELL_TYPE_NUMERIC) {
                almvc1 = balanceCell.getNumericCellValue();
            } else {
                almvc1 = Double.parseDouble(balanceCell.toString());
            }

            // Apply filtering conditions
            if (bank.equals("SABB") &&
                product.equals("CUT") &&
                currency.equals("SAR") &&
                almvc1 > 0) {

                // Convert Halalas to SAR
                double balance = almvc1 / 100;

                // Add record to list
                records.add(new Record(aldcs, balance));
            }
        }

        // Close workbook and file
        wb.close();
        fis.close();

        // Return filtered records
        return records;
    }
}
📄 4. CalculationService.java
import java.util.*;

// This class contains all calculation logic
public class CalculationService {

    // Group data by ALDCS category
    public Map groupByCategory(List records) {

        Map map = new HashMap();

        // Loop all records
        for (Record r : records) {

            // Add balance per category
            if (!map.containsKey(r.aldcs)) {
                map.put(r.aldcs, r.balance);
            } else {
                map.put(r.aldcs, map.get(r.aldcs) + r.balance);
            }
        }

        return map;
    }

    // Perform full NII calculation
    public Map calculate(Map totals, InputParams p) {

        Map result = new HashMap();

        // Total balance
        double total = 0;
        for (double v : totals.values()) {
            total += v;
        }

        // Stat reserve deduction
        double statReserve = total * p.statReserve;

        // Eligible balance
        double eligible = total - statReserve;

        // VOF calculation
        double vof = eligible * (p.vofRate / 100) * p.days / 365;

        // DPF calculation
        double dpf = eligible * (p.dpfRate / 100) * p.days / 365;

        // HSS share
        double hss = vof - dpf;

        // NII calculation
        double nii = total * p.repoRate * p.days / 360;

        // Adjusted NII
        double niiAdj = nii * p.niiFactor;

        // VAT
        double vat = niiAdj * p.vatRate;

        // Final claim
        double claim = niiAdj + vat;

        // Store all values
        result.put("TOTAL", total);
        result.put("ELIGIBLE", eligible);
        result.put("VOF", vof);
        result.put("DPF", dpf);
        result.put("HSS", hss);
        result.put("NII", niiAdj);
        result.put("VAT", vat);
        result.put("CLAIM", claim);

        return result;
    }
}

import org.apache.poi.xssf.usermodel.*;
import java.io.FileOutputStream;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.Map;

// This class generates output Excel
public class ReportService {

    public String generate(Map r) throws Exception {

        // Create workbook
        XSSFWorkbook wb = new XSSFWorkbook();

        // Create sheet
        XSSFSheet sheet = wb.createSheet("Working");

        int row = 0;

        // Title
        sheet.createRow(row++).createCell(0).setCellValue("NII ESTIMATION");

        row++;

        // Write values
        sheet.createRow(row).createCell(0).setCellValue("Total Balance");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("TOTAL"));

        sheet.createRow(row).createCell(0).setCellValue("Eligible Balance");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("ELIGIBLE"));

        sheet.createRow(row).createCell(0).setCellValue("NII");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("NII"));

        sheet.createRow(row).createCell(0).setCellValue("VAT");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("VAT"));

        sheet.createRow(row).createCell(0).setCellValue("Claim");
        sheet.getRow(row++).createCell(1).setCellValue(r.get("CLAIM"));

        // Generate file name with timestamp
        String time = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());

        String path = "output/NII_Output_" + time + ".xlsx";

        // Write file
        FileOutputStream out = new FileOutputStream(path);
        wb.write(out);

        // Close
        wb.close();
        out.close();

        return path;
    }
}

// This class creates backup of input and output
public class BackupService {

    public void backup(String input, String output) throws Exception {

        // Generate timestamp
        String time = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());

        // Create backup folder if not exists
        File dir = new File("bkp");
        if (!dir.exists()) dir.mkdir();

        // Copy files
        Files.copy(Paths.get(input), Paths.get("bkp/input_" + time + ".xlsx"));
        Files.copy(Paths.get(output), Paths.get("bkp/output_" + time + ".xlsx"));
    }
}
📄 7. NiiController.java
import java.util.*;

// This class controls full execution
public class NiiController {

    ExcelReaderService reader = new ExcelReaderService();
    CalculationService calc = new CalculationService();
    ReportService report = new ReportService();
    BackupService backup = new BackupService();

    public void execute(String sourceFile, InputParams params) throws Exception {

        // Step 1: Read input file
        List records = reader.readSource(sourceFile);

        // Step 2: Group data
        Map grouped = calc.groupByCategory(records);

        // Step 3: Perform calculation
        Map result = calc.calculate(grouped, params);

        // Step 4: Generate report
        String output = report.generate(result);

        // Step 5: Backup
        backup.backup(sourceFile, output);
    }
}
