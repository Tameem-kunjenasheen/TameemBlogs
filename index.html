package mainUI;

import java.io.*;
import java.text.SimpleDateFormat;
import java.util.Date;

// Backup input & output
public class BackupService {

    public void backup(String input, String output) throws Exception {

        String time = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());

        copy(input, "backup/input_" + time + ".xlsx");
        copy(output, "backup/output_" + time + ".xlsx");
    }

    private void copy(String src, String dest) throws Exception {

        InputStream in = new FileInputStream(src);
        OutputStream out = new FileOutputStream(dest);

        byte[] buffer = new byte[1024];
        int len;

        while ((len = in.read(buffer)) > 0) {
            out.write(buffer, 0, len);
        }

        in.close();
        out.close();
    }
}

import java.io.File;

public class ExcelReaderService {

    public List read(String path) throws Exception {

        List list = new ArrayList<>();

        Workbook wb = WorkbookFactory.create(new File(path));
        Sheet sheet = wb.getSheet("Source");

        Map headerMap = new HashMap<>();

        Row header = sheet.getRow(0);

        // Dynamic column mapping
        for (Cell cell : header) {
            headerMap.put(cell.getStringCellValue().trim(), cell.getColumnIndex());
        }

        for (Row row : sheet) {

            if (row.getRowNum() == 0) continue;

            String product = row.getCell(headerMap.get("ALPRIE")).toString();
            String currency = row.getCell(headerMap.get("ALCYCD")).toString();
            String category = row.getCell(headerMap.get("ALDCS")).toString();
            String halal = row.getCell(headerMap.get("ALCTCD")).toString();

            double balance = row.getCell(headerMap.get("ALMVC1")).getNumericCellValue() / 100;

            if (product.equals("CUT") && currency.equals("SAR")) {

                list.add(new Record(category, balance, halal.equals("SA")));
            }
        }

        wb.close();
        return list;
    }
}

            if (r.isHalal()) {

        result.put("TOTAL", total);
        result.put("ELIGIBLE", eligible);
        result.put("VOF", vof);
        result.put("DPF", dpf);
        result.put("HSS", hss);
