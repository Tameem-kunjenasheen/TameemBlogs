package mainUI;

import org.apache.poi.ss.usermodel.*;
import java.io.File;
import java.util.*;

// Reads input Excel file
public class ExcelReaderService {

    public List readSource(String filePath) throws Exception {

        List list = new ArrayList();

        Workbook wb = WorkbookFactory.create(new File(filePath));
        Sheet sheet = wb.getSheetAt(0);

        for (Row row : sheet) {

            // Skip header
            if (row.getRowNum() == 0) continue;

            // Read ALDCS (column C → index 2)
            Cell c1 = row.getCell(2);

            // Read value (column K → index 10)
            Cell c2 = row.getCell(10);

            if (c1 == null || c2 == null) continue;

            String aldcs = c1.toString();
            double value = c2.getNumericCellValue();

            list.add(new Record(aldcs, value));
        }

        wb.close();

        return list;
    }
}

    public Map calculate(Map totals, InputParams p) {

        Map result = new HashMap();

        double total = 0;

        for (double v : totals.values()) {
            total += v;
        }

        double statReserve = total * p.statReserve;
        double eligible = total - statReserve;

        double vof = eligible * (p.vofRate / 100) * p.days / 365;
        double dpf = eligible * (p.dpfRate / 100) * p.days / 365;
        double hss = vof - dpf;

        double nii = total * p.repoRate * p.days / 360;
        double niiAdj = nii * p.niiFactor;

        double vat = niiAdj * p.vatRate;
        double claim = niiAdj + vat;

        result.put("TOTAL", total);
        result.put("STAT_RESERVE", statReserve);
        result.put("ELIGIBLE", eligible);
        result.put("VOF", vof);
        result.put("DPF", dpf);
        result.put("HSS", hss);

package mainUI;

import java.io.*;
import java.text.SimpleDateFormat;
import java.util.Date;

// Backup input & output
public class BackupService {

    public void backup(String input, String output) throws Exception {

        String time = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());

        copy(input, "backup/input_" + time + ".xlsx");
        copy(output, "backup/output_" + time + ".xlsx");
    }

    private void copy(String src, String dest) throws Exception {

        InputStream in = new FileInputStream(src);
        OutputStream out = new FileOutputStream(dest);

        byte[] buffer = new byte[1024];
        int len;

        while ((len = in.read(buffer)) > 0) {
            out.write(buffer, 0, len);
        }

        in.close();
        out.close();
    }
}
